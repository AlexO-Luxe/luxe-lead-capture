<script>
/* ══════════════════════════════════════════════════════════
   TAB SWAP LOGIC
   Nav in hero swaps between #slrPanelForm and #slrPanelSpeak
   without scrolling the page. Hero title/sub also swap.
   Deep links: #speak-to-us / #enquiry-form supported.
   ══════════════════════════════════════════════════════════ */
(function(){
  var panels = {
    form:  document.getElementById('slrPanelForm'),
    speak: document.getElementById('slrPanelSpeak')
  };
  var btns = {
    form:  document.getElementById('slrTabForm'),
    speak: document.getElementById('slrTabSpeak')
  };
  var title = document.getElementById('slrHeroTitle');
  var sub   = document.getElementById('slrHeroSub');

  var COPY = {
    form: {
      title: 'Availability &amp; options <em>on request</em>',
      sub:   'Fill in the form and one of our accommodation experts will handle your enquiry.'
    },
    speak: {
      title: 'Get in touch with<br>our <em>Head Office.</em>',
      sub:   'However you reach us, you get the same London team, no call centre, no ticket queue. Pick the channel that fits your question.'
    }
  };

  function setTab(tab, updateHash){
    if(!panels[tab]) return;
    Object.keys(panels).forEach(function(k){
      if(panels[k]) panels[k].classList.toggle('slr-panel-active', k === tab);
      if(btns[k]){
        btns[k].classList.toggle('active', k === tab);
        btns[k].setAttribute('aria-selected', k === tab ? 'true' : 'false');
      }
    });
    if(title && COPY[tab]) title.innerHTML = COPY[tab].title;
    if(sub   && COPY[tab]) sub.innerHTML   = COPY[tab].sub;
    if(updateHash && history && history.replaceState){
      var hash = tab === 'speak' ? '#speak-to-us' : '#enquiry-form';
      history.replaceState(null, '', hash);
    }
  }

  Object.keys(btns).forEach(function(k){
    if(btns[k]) btns[k].addEventListener('click', function(){ setTab(k, true); });
  });

  // Any element with data-slr-tab triggers a tab switch (used by the back button)
  document.querySelectorAll('[data-slr-tab]').forEach(function(el){
    el.addEventListener('click', function(){ setTab(el.getAttribute('data-slr-tab'), true); });
  });

  // Deep-link support on load
  if(location.hash === '#speak-to-us') setTab('speak', false);
  window.addEventListener('hashchange', function(){
    if(location.hash === '#speak-to-us')  setTab('speak', false);
    if(location.hash === '#enquiry-form') setTab('form',  false);
  });

  // Mirror the form's response-status pill into the concierge panel
  function syncCxStatus(){
    var srcPill = document.querySelector('.sl-enq .sl-status-pill');
    var srcDot  = document.querySelector('.sl-enq .sl-status-dot');
    if(!srcPill && !srcDot) return;
    var dstPill = document.getElementById('slrCxStatusPill');
    var dstDot  = document.getElementById('slrCxStatusDot');
    var dstText = document.getElementById('slrCxStatusText');
    if(srcPill && dstPill){
      dstPill.textContent = srcPill.textContent;
      dstPill.classList.toggle('amber', srcPill.classList.contains('amber'));
    }
    if(srcDot && dstDot){
      dstDot.classList.toggle('amber', srcDot.classList.contains('amber'));
    }
    // Amber = closure. Swap the copy so both panels stay in sync.
    if(dstText && srcPill && srcPill.classList.contains('amber')){
      dstText.textContent = 'Reduced hours right now. Reply times may be slower than usual.';
    }
  }
  // Response status is set on the form after DOM is ready. Give it a beat.
  setTimeout(syncCxStatus, 300);
})();
</script>

  function getDialData(){
    var opt = dialSel.options[dialSel.selectedIndex];
    var d   = parseInt(opt.getAttribute('data-d'),10) || 9;
    var dmin = opt.getAttribute('data-dmin') ? parseInt(opt.getAttribute('data-dmin'),10) : d;
    var dmax = opt.getAttribute('data-dmax') ? parseInt(opt.getAttribute('data-dmax'),10) : d;
    return {
      code:   dialSel.value || '',
      digits: d,
      dmin:   dmin,
      dmax:   dmax,
      ph:     opt.getAttribute('data-ph') || ''
    };
  }

  function updatePlaceholder(){
    var d = getDialData();
    phoneInp.placeholder = d.ph || 'Enter phone number';
    dialPrefix.textContent = d.code || '+';
  }

  // Auto-strip leading zero for dial codes that don't use them (e.g. +44)
  // Countries where a leading 0 is expected in local format (trunk prefix):
  var TRUNK_PREFIX_CODES = ['+44','+33','+49','+39','+34','+31','+32','+41','+43','+46','+47','+45','+353','+61','+64','+351','+48','+30','+90','+380','+40','+36','+420','+358','+370','+371','+372','+381','+385','+386','+421','+355','+359','+373','+382','+387','+992','+993','+994','+995','+996','+998','+60','+62','+63','+66','+84','+95','+94','+92','+880','+977'];

  var zeroStripHint = document.getElementById('slPhoneZeroHint');
  var phoneHintSep  = document.getElementById('slPhoneHintSep');

  function updateHintSep(){
    if(phoneHintSep){
      var zeroVisible = zeroStripHint && zeroStripHint.style.display !== 'none';
      var hintVisible = phoneHint.textContent.trim().length > 0;
      phoneHintSep.style.display = (zeroVisible && hintVisible) ? 'inline' : 'none';
    }
  }

  function stripLeadingZero(){
    var code = dialSel.value;
    if(TRUNK_PREFIX_CODES.indexOf(code) === -1) return; // not applicable
    var val = phoneInp.value.replace(/\s/g,'');
    if(val.charAt(0) === '0'){
      phoneInp.value = phoneInp.value.replace(/^(\s*)0/, '$1'); // strip leading 0, preserve spacing
      if(zeroStripHint){
        zeroStripHint.style.display = 'flex';
        clearTimeout(zeroStripHint._t);
        zeroStripHint._t = setTimeout(function(){ zeroStripHint.style.display = 'none'; updateHintSep(); }, 3000);
        updateHintSep();
      }
    }
  }

  function stripDigits(v){ return v.replace(/\D/g,''); }

  function updateHint(digits, required, touched, dmin, dmax){
    dmin = dmin || required; dmax = dmax || required;
    if(!touched || digits.length === 0){
      phoneHint.textContent = '';
      phoneHint.className = 'sl-phone-digits-hint';
      updateHintSep();
      return;
    }
    var len = digits.length;
    if(len < dmin){
      var rem = dmin - len;
      phoneHint.textContent = rem + ' more digit' + (rem===1?'':'s') + ' needed';
      phoneHint.className = 'sl-phone-digits-hint sl-warn';
    } else if(len > dmax){
      var over = len - dmax;
      phoneHint.textContent = over + ' digit' + (over===1?'':'s') + ' too many — please remove ' + over;
      phoneHint.className = 'sl-phone-digits-hint sl-warn';
    } else {
      phoneHint.textContent = '✓ Looks good';
      phoneHint.className = 'sl-phone-digits-hint sl-good';
    }
    updateHintSep();
  }

  function validatePhone(showE){
    var d      = getDialData();
    var digits = stripDigits(phoneInp.value);
    var len    = digits.length;
    var valid  = len >= d.dmin && len <= d.dmax;
    var touched= phoneInp.value.length > 0;
    setFs(phoneWrap, valid ? 'ok' : (showE && touched ? 'err' : null));
    if(showE && !valid && touched){
      // Show error message and clear inline hint to avoid duplication
      showErr('slErr-phone', true);
      var msg = document.getElementById('slErr-phone');
      if(len < d.dmin){
        var rem = d.dmin - len;
        msg.textContent = 'Too short — ' + rem + ' more digit'+(rem===1?'':'s')+' needed for '+d.code;
      } else if(len > d.dmax){
        var over = len - d.dmax;
        msg.textContent = over + ' digit'+(over===1?'':'s')+' too many — please remove '+over;
      } else {
        msg.textContent = 'Please enter a valid phone number';
      }
      phoneHint.textContent = '';
      phoneHint.className = 'sl-phone-digits-hint';
    } else {
      if(showE) showErr('slErr-phone', false);
      updateHint(digits, d.digits, touched, d.dmin, d.dmax);
    }
    return valid;
  }

  // Strip non-numeric (allow spaces/dashes for readability), live hint
  phoneInp.addEventListener('input', function(){
    this.value = this.value.replace(/[^\d\s\-().+]/g,'');
    // Auto-strip leading zero for +44 (UK) numbers
    stripLeadingZero();
    var d = getDialData();
    updateHint(stripDigits(this.value), d.digits, this.value.length > 0, d.dmin, d.dmax);
    setFs(phoneWrap, null); // reset border while typing
    showErr('slErr-phone', false);
  });

    if(this.value.length>0) validatePhone(true);

  // ── SEARCHABLE DIAL DROPDOWN ──────────────────────────────────────
  (function(){
    var trigger    = document.getElementById('slDialTrigger');
    var dropdown   = document.getElementById('slDialDropdown');
    var searchInp  = document.getElementById('slDialSearch');
    var list       = document.getElementById('slDialList');
    var noResults  = document.getElementById('slDialNoResults');
    var opts       = list ? Array.from(list.querySelectorAll('.sl-dial-opt')) : [];
    var isOpen     = false;

    function openDrop(){
      isOpen = true;
      dropdown.classList.add('sl-open');
      trigger.classList.add('sl-open');
      trigger.setAttribute('aria-expanded','true');
      searchInp.value = '';
      filterOpts('');
      setTimeout(function(){ searchInp.focus(); }, 10);
    }

    function closeDrop(){
      isOpen = false;
      dropdown.classList.remove('sl-open');
      trigger.classList.remove('sl-open');
      trigger.setAttribute('aria-expanded','false');
    }

    function filterOpts(q){
      q = q.toLowerCase().trim();

      var any = false;
      opts.forEach(function(opt){
        var label = (opt.getAttribute('data-label') || opt.textContent).toLowerCase();
        var val   = (opt.getAttribute('data-value') || '').toLowerCase();
        var show  = !q || label.indexOf(q) > -1 || val.indexOf(q) > -1;
        opt.classList.toggle('sl-hidden', !show);
        if(show) any = true;
      });
      // Hide group labels if all their options are hidden
      list.querySelectorAll('.sl-dial-group-label').forEach(function(gl){
        var next = gl.nextElementSibling;
        var hasVisible = false;
        while(next && !next.classList.contains('sl-dial-group-label')){
          if(next.classList.contains('sl-dial-opt') && !next.classList.contains('sl-hidden')){
            hasVisible = true; break;
          }
          next = next.nextElementSibling;
        }
        gl.style.display = hasVisible ? '' : 'none';
      });
      noResults.style.display = any ? 'none' : '';
    }

    function selectOpt(opt){
      var val   = opt.getAttribute('data-value');
      var label = opt.getAttribute('data-label') || opt.textContent;

      // Sync to hidden select
      for(var i = 0; i < dialSel.options.length; i++){
        if(dialSel.options[i].value === val){
          // Match on value AND data-d to handle duplicate values like +1 (US/Canada)
          var selOpt = dialSel.options[i];
          var samePh = selOpt.getAttribute('data-ph') === opt.getAttribute('data-ph');
          if(samePh || dialSel.options[i].value === val){
            dialSel.selectedIndex = i;
            // Keep looking if +1 but wrong country
            if(selOpt.textContent.trim() === label.trim()) break;
          }
        }
      }

      // Update trigger text
      trigger.textContent = label;
      trigger.style.color = '';
      trigger.classList.add('sl-filled');

      // Mark selected state
      opts.forEach(function(o){ o.classList.remove('sl-selected'); });
      opt.classList.add('sl-selected');

      closeDrop();

      // Fire change on hidden select so existing listeners run
      dialSel.dispatchEvent(new Event('change'));
    }

    if(trigger){
      trigger.addEventListener('click', function(e){
        e.stopPropagation();
        isOpen ? closeDrop() : openDrop();
      });
    }

    if(searchInp){
      searchInp.addEventListener('input', function(){
        filterOpts(this.value);
      });
      // Prevent dropdown close when clicking search
      searchInp.addEventListener('mousedown', function(e){ e.stopPropagation(); });
    }

    // Option click
    opts.forEach(function(opt){
      opt.addEventListener('mousedown', function(e){
        e.preventDefault();
        selectOpt(opt);
      });
    });

    // Close on outside click
    document.addEventListener('click', function(e){
      if(isOpen && !document.getElementById('slDialWrap').contains(e.target)){
        closeDrop();
      }
    });

    // Keyboard: Escape closes
    document.addEventListener('keydown', function(e){
      if(isOpen && e.key === 'Escape') closeDrop();
    });

    // No default selection — user must choose

  })();

      if(!target) return;
      var rect = target.getBoundingClientRect();
      var scrollTop = window.pageYOffset || document.documentElement.scrollTop;
      var isMobile = window.innerWidth < 600;
      if(isMobile){
        var viewH = window.innerHeight || document.documentElement.clientHeight;
        if(rect.bottom > viewH - 40){
          window.scrollTo({ top: scrollTop + rect.bottom - viewH + 80, behavior: 'smooth' });
        }
      } else {
        window.scrollTo({ top: scrollTop + rect.top - 340, behavior: 'smooth' });
      }

  function getBudgetPeriod(city){
    var tier = CITY_TIER[city] || '';
    if(tier.indexOf('uk') === 0) return 'week';
    return 'month';

  function rebuildBudget(){
    var city  = cityEl2.value;
    var apt   = aptTypeEl.value || 'studio';
    var tiers = getTiers(city, apt);
    var period = getBudgetPeriod(city);

    // Update label
    if(budgetLabelEl) budgetLabelEl.innerHTML = 'Budget per ' + period + ' <span class="sl-req">*</span>';

    budgetEl.innerHTML = '<option value="">Select budget\u2026</option>';
    budgetSorry.classList.remove('sl-show');
    enqWrap.classList.remove('sl-locked');

    if(!tiers){ return; }

    tiers.forEach(function(t){
      var opt = document.createElement('option');
      opt.value = t.value;
      opt.textContent = t.label;
      budgetEl.appendChild(opt);
    });
  }

  function checkBudgetLockout(){
    var city  = cityEl2.value;
    var apt   = aptTypeEl.value || 'studio';
    var tiers = getTiers(city, apt);
    var tier  = null;
    var tierIndex = -1;
    if(tiers) tiers.forEach(function(t, i){ if(t.value===budgetEl.value){ tier=t; tierIndex=i; } });
    var locked = !!(tier && tier.lockout);
    if(locked){
      // Apartment type label
      var aptNames = {
        'studio':'studio apartments', '1bed':'1 bedroom apartments',
        '2bed':'2 bedroom apartments', '3bed':'3 bedroom apartments',
        '4bed':'4 bedroom apartments', 'flexible':'apartments'
      };
      var aptLabel = aptNames[apt] || 'apartments';

      // Find the starting price from the first non-lockout tier
      var startLabel = '';
      if(tiers && tierIndex >= 0){
        for(var i = tierIndex + 1; i < tiers.length; i++){
          if(!tiers[i].lockout){
            // Extract just the first price from the label e.g. "£650 – £1,000" → "£650"
            var match = tiers[i].label.match(/^([£€\$][0-9,]+)/);
            if(match) startLabel = match[1];
            break;
          }
        }
      }

      var fromText = startLabel ? ' from ' + startLabel + ' per ' + getBudgetPeriod(city) : '';

  function updateStayDuration(ciVal, coVal){
    if(!ciVal || !coVal){

      stayMinSorry.style.display = 'none';
      enqWrap.classList.remove('sl-stay-locked');
      return;
    }
    var weeks   = Math.floor(nights / 7);
    var days    = nights % 7;
    var readable = weeks > 0 && days > 0 ? weeks+' week'+(weeks>1?'s':'')+' and '+days+' night'+(days>1?'s':'')
                 : weeks > 0 ? weeks+' week'+(weeks>1?'s':'')
                 : nights+' night'+(nights>1?'s':'');

  function validateEmail(doShow){
    var emailEl   = document.getElementById('slEmail');
    var emailErr  = document.getElementById('slErr-email');
    var val       = emailEl.value.trim().toLowerCase();
    var hintEl    = document.getElementById('slEmailDomainHint');

    if(hintEl) hintEl.style.display = 'none';

    if(!val){
      if(doShow){ showErr('slErr-email', true); emailErr.textContent = 'Please enter your email address'; setFs(emailEl,'err'); }
      return false;
    }

    if(!/^[^\s@]+@[^\s@]+\.[^\s@]{2,}$/.test(val)){
      if(doShow){ showErr('slErr-email', true); emailErr.textContent = 'Please enter a valid email address'; setFs(emailEl,'err'); }
      return false;
    }

    var parts  = val.split('@');
    var prefix = parts[0];
    var domain = parts[1];

    if(TYPO_MAP[domain] && hintEl){
      var hintSpan = hintEl.querySelector('span');
      if(hintSpan) hintSpan.textContent = 'Did you mean ' + prefix + '@' + TYPO_MAP[domain] + '? Tap to fix';
      hintEl.style.display = 'flex';
      // Remove tick while domain hint is showing
      var emailField = emailEl.closest ? emailEl.closest('.sl-field') : null;
      if(emailField) emailField.removeAttribute('data-filled');
      hintEl.onclick = function(){
        emailEl.value = prefix + '@' + TYPO_MAP[domain];
        hintEl.style.display = 'none';
        applyFilled(emailEl, true);
        setFs(emailEl,'ok');
        showErr('slErr-email', false);
      };
    }

    if(DISPOSABLE_DOMAINS.indexOf(domain) > -1){
      if(doShow){ showErr('slErr-email', true); emailErr.textContent = 'Please use a personal email address — temporary inboxes aren\'t accepted'; setFs(emailEl,'err'); }
      return false;
    }

    if(ROLE_PREFIXES.indexOf(prefix) > -1){
      if(doShow){ showErr('slErr-email', true); emailErr.textContent = 'Please use your personal email address rather than a shared inbox'; setFs(emailEl,'err'); }
      return false;
    }

    if(doShow){ showErr('slErr-email', false); setFs(emailEl,'ok'); }

  function applyFilled(el, onBlur){
    if(!el) return;

    el.classList.toggle('sl-filled', !!filled);

    if(!field){ var p = el.parentNode; while(p){ if(p.classList && p.classList.contains('sl-field')){ field=p; break; } p=p.parentNode; } }
    if(field){
      // Only show tick on blur (onBlur=true) or polling, and never when errors/warnings present
      var hasErr  = el.classList.contains('sl-err');

      var hasEmailHint = field.querySelector('[id$="DomainHint"]') && field.querySelector('[id$="DomainHint"]').style.display !== 'none';

function runPrefill () {
  var params = new URLSearchParams(window.location.search);
  function get (key) { return params.get(key); }

  window.slPrefilling = true;

  var type = get('type');
  if (type === 'A' || type === 'B') {
    var radio = document.querySelector('input[name="enquiryType"][value="' + type + '"]');
    if (radio) { radio.checked = true; radio.dispatchEvent(new Event('change')); }
  }

  var ref = get('ref');
  if (ref) {
    var refEl = document.getElementById('slAptRef');
    if (refEl) {
      if (!get('type')) {
        var radioA = document.querySelector('input[name="enquiryType"][value="A"]');
        if (radioA) { radioA.checked = true; radioA.dispatchEvent(new Event('change')); }
      }
      refEl.value = ref;
      applyFilled(refEl);
      setTimeout(function () { applyFilled(refEl); }, 50);
    }
  }

  var city = get('city');
  if (city) {
    var cityEl = document.getElementById('slCity');
    if (cityEl) { cityEl.value = city; cityEl.dispatchEvent(new Event('change')); applyFilled(cityEl); }
  }

  var apt = get('apt');
  if (apt) {
    var aptEl = document.getElementById('slAptType');
    if (aptEl) { aptEl.value = apt; aptEl.dispatchEvent(new Event('change')); applyFilled(aptEl); }
  }

  var name = get('name');
  if (name) {
    var nameEl = document.getElementById('slFullName');
    if (nameEl) { nameEl.value = name; applyFilled(nameEl); }
  }

  var email = get('email');
  if (email) {
    var emailEl = document.getElementById('slEmail');
    if (emailEl) { emailEl.value = email; applyFilled(emailEl); }
  }

  var checkin  = get('checkin');
  var checkout = get('checkout');
  if (checkin || checkout) {
    var attempts = 0;
    var wait = setInterval(function () {
      attempts++;
      var fpIn  = window._fpIn;
      var fpOut = window._fpOut;
      if ((fpIn && fpOut) || attempts > 20) {
        clearInterval(wait);
        if (checkin  && fpIn)  fpIn.setDate(checkin,  true);
        if (checkout && fpOut) fpOut.setDate(checkout, true);
      }
    }, 150);
  }

  window.slPrefilling = false;

  var uniParam = get('university');
  if (uniParam) {
    var uniEl = document.getElementById('slUniversity');
    if (uniEl) {
      var matched = false;
      for (var i = 0; i < uniEl.options.length; i++) {
        if (uniEl.options[i].value.toLowerCase() === uniParam.toLowerCase()) {
          uniEl.value = uniEl.options[i].value;
          applyFilled(uniEl);
          matched = true;
          break;
        }
      }
      if (!matched) {
        uniEl.value = 'Other';
        uniEl.dispatchEvent(new Event('change'));
        applyFilled(uniEl);
        var otherUniEl = document.getElementById('slOtherUni');
        if (otherUniEl) { otherUniEl.value = uniParam; applyFilled(otherUniEl); }
      }
    }
  }

  if ((type === 'A' || type === 'B') && typeof scrollToNextField === 'function') {
    setTimeout(function () { scrollToNextField(type === 'A'); }, 500);
  }
}
