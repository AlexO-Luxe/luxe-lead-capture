  function getDialData(){

  var TRUNK_PREFIX_CODES = ['+44','+33','+49','+39','+34','+31','+32','+41','+43','+46','+47','+45','+353','+61','+64','+351','+48','+30','+90','+380','+40','+36','+420','+358','+370','+371','+372','+381','+385','+386','+421','+355','+359','+373','+382','+387','+992','+993','+994','+995','+996','+998','+60','+62','+63','+66','+84','+95','+94','+92','+880','+977'];

  function updateHintSep(){
    if(phoneHintSep){
      var zeroVisible = zeroStripHint && zeroStripHint.style.display !== 'none';
      var hintVisible = phoneHint.textContent.trim().length > 0;
      phoneHintSep.style.display = (zeroVisible && hintVisible) ? 'inline' : 'none';

  function stripLeadingZero(){
    var code = dialSel.value;
    if(TRUNK_PREFIX_CODES.indexOf(code) === -1) return; // not applicable
    var val = phoneInp.value.replace(/\s/g,'');
    if(val.charAt(0) === '0'){
      phoneInp.value = phoneInp.value.replace(/^(\s*)0/, '$1'); // strip leading 0, preserve spacing
      if(zeroStripHint){
        zeroStripHint.style.display = 'flex';
        clearTimeout(zeroStripHint._t);
        zeroStripHint._t = setTimeout(function(){ zeroStripHint.style.display = 'none'; updateHintSep(); }, 3000);
        updateHintSep();
      }
    }
  }

  function stripDigits(v){ return v.replace(/\D/g,''); }

  function updateHint(digits, required, touched, dmin, dmax){
    dmin = dmin || required; dmax = dmax || required;
    if(!touched || digits.length === 0){
      phoneHint.textContent = '';
      phoneHint.className = 'sl-phone-digits-hint';
      updateHintSep();
      return;
    }
    var len = digits.length;
    if(len < dmin){
      var rem = dmin - len;
      phoneHint.textContent = rem + ' more digit' + (rem===1?'':'s') + ' needed';
      phoneHint.className = 'sl-phone-digits-hint sl-warn';
    } else if(len > dmax){
      var over = len - dmax;
      phoneHint.textContent = over + ' digit' + (over===1?'':'s') + ' too many вЂ” please remove ' + over;
      phoneHint.className = 'sl-phone-digits-hint sl-warn';
    } else {
      phoneHint.textContent = 'вњ“ Looks good';
      phoneHint.className = 'sl-phone-digits-hint sl-good';
    }
    updateHintSep();
  }

  function validatePhone(showE){
    var d      = getDialData();
    var digits = stripDigits(phoneInp.value);
    var len    = digits.length;
    var valid  = len >= d.dmin && len <= d.dmax;
    var touched= phoneInp.value.length > 0;
    setFs(phoneWrap, valid ? 'ok' : (showE && touched ? 'err' : null));
    if(showE && !valid && touched){
      // Show error message and clear inline hint to avoid duplication
      showErr('slErr-phone', true);
      var msg = document.getElementById('slErr-phone');
      if(len < d.dmin){
        var rem = d.dmin - len;
        msg.textContent = 'Too short вЂ” ' + rem + ' more digit'+(rem===1?'':'s')+' needed for '+d.code;
      } else if(len > d.dmax){
        var over = len - d.dmax;
        msg.textContent = over + ' digit'+(over===1?'':'s')+' too many вЂ” please remove '+over;
      } else {
        msg.textContent = 'Please enter a valid phone number';
      }
      phoneHint.textContent = '';
      phoneHint.className = 'sl-phone-digits-hint';
    } else {
      if(showE) showErr('slErr-phone', false);
      updateHint(digits, d.digits, touched, d.dmin, d.dmax);
    }
    return valid;
  }

    function filterOpts(q){
      q = q.toLowerCase().trim();

      var any = false;
      opts.forEach(function(opt){
        var label = (opt.getAttribute('data-label') || opt.textContent).toLowerCase();
        var val   = (opt.getAttribute('data-value') || '').toLowerCase();
        var show  = !q || label.indexOf(q) > -1 || val.indexOf(q) > -1;
        opt.classList.toggle('sl-hidden', !show);
        if(show) any = true;
      });
      // Hide group labels if all their options are hidden
      list.querySelectorAll('.sl-dial-group-label').forEach(function(gl){
        var next = gl.nextElementSibling;
        var hasVisible = false;
        while(next && !next.classList.contains('sl-dial-group-label')){
          if(next.classList.contains('sl-dial-opt') && !next.classList.contains('sl-hidden')){
            hasVisible = true; break;
          }
          next = next.nextElementSibling;
        }
        gl.style.display = hasVisible ? '' : 'none';
      });

      var val   = opt.getAttribute('data-value');
      var label = opt.getAttribute('data-label') || opt.textContent;

      // Sync to hidden select
      for(var i = 0; i < dialSel.options.length; i++){
        if(dialSel.options[i].value === val){
          // Match on value AND data-d to handle duplicate values like +1 (US/Canada)
          var selOpt = dialSel.options[i];
          var samePh = selOpt.getAttribute('data-ph') === opt.getAttribute('data-ph');
          if(samePh || dialSel.options[i].value === val){
            dialSel.selectedIndex = i;
            // Keep looking if +1 but wrong country
            if(selOpt.textContent.trim() === label.trim()) break;
          }
        }
      }

      trigger.textContent = label;
      trigger.style.color = '';
      trigger.classList.add('sl-filled');

      // Mark selected state
      opts.forEach(function(o){ o.classList.remove('sl-selected'); });
      opt.classList.add('sl-selected');

  function getTiers(city, apt){
    var tier = CITY_TIER[city];

  function checkBudgetLockout(){
    var city  = cityEl2.value;
    var apt   = aptTypeEl.value || 'studio';
    var tiers = getTiers(city, apt);
    var tier  = null;
    var tierIndex = -1;
    if(tiers) tiers.forEach(function(t, i){ if(t.value===budgetEl.value){ tier=t; tierIndex=i; } });
    var locked = !!(tier && tier.lockout);
    if(locked){
      // Apartment type label
      var aptNames = {
        'studio':'studio apartments', '1bed':'1 bedroom apartments',
        '2bed':'2 bedroom apartments', '3bed':'3 bedroom apartments',
        '4bed':'4 bedroom apartments', 'flexible':'apartments'
      };
      var aptLabel = aptNames[apt] || 'apartments';

      // Find the starting price from the first non-lockout tier
      var startLabel = '';
      if(tiers && tierIndex >= 0){
        for(var i = tierIndex + 1; i < tiers.length; i++){
          if(!tiers[i].lockout){
            // Extract just the first price from the label e.g. "ВЈ650 вЂ“ ВЈ1,000" в†’ "ВЈ650"
            var match = tiers[i].label.match(/^([ВЈв‚¬\$][0-9,]+)/);
            if(match) startLabel = match[1];
            break;
          }
        }
      }

      var fromText = startLabel ? ' from ' + startLabel + ' per ' + getBudgetPeriod(city) : '';
      budgetSorry.textContent = 'Apologies \u2014 our ' + aptLabel + ' start' + fromText + '. Remember \u2014 all our prices include bills, housekeeping, utilities \u2014 everything.';
    }
    budgetSorry.classList.toggle('sl-show', locked);
    enqWrap.classList.toggle('sl-locked', locked);
    if(locked) budgetSorry.scrollIntoView({behavior:'smooth', block:'nearest'});
  }

  function updateStayDuration(ciVal, coVal){
    if(!ciVal || !coVal){
      stayDurationWrap.style.display = 'none';
      stayMinSorry.style.display = 'none';
      enqWrap.classList.remove('sl-stay-locked');
      return;
    }
    var start  = new Date(ciVal);
    var end    = new Date(coVal);
    var nights = Math.round((end - start) / (1000 * 60 * 60 * 24));
    if(nights <= 0){
      stayDurationWrap.style.display = 'none';
      stayMinSorry.style.display = 'none';
      enqWrap.classList.remove('sl-stay-locked');
      return;
    }
    var weeks   = Math.floor(nights / 7);
    var days    = nights % 7;
    var readable = weeks > 0 && days > 0 ? weeks+' week'+(weeks>1?'s':'')+' and '+days+' night'+(days>1?'s':'')
                 : weeks > 0 ? weeks+' week'+(weeks>1?'s':'')
                 : nights+' night'+(nights>1?'s':'');
    var tooShort = nights < MIN_NIGHTS;
    stayDurationWrap.style.display = tooShort ? 'none' : 'flex';
    if(!tooShort) stayDurationText.textContent = 'You\'re staying ' + nights + ' night' + (nights === 1 ? '' : 's');
    stayMinSorry.style.display = tooShort ? 'block' : 'none';
    enqWrap.classList.toggle('sl-stay-locked', tooShort);
  }

  function validateEmail(doShow){
    var emailEl   = document.getElementById('slEmail');
    var emailErr  = document.getElementById('slErr-email');
    var val       = emailEl.value.trim().toLowerCase();
    var hintEl    = document.getElementById('slEmailDomainHint');

    if(hintEl) hintEl.style.display = 'none';

    if(!val){
      if(doShow){ showErr('slErr-email', true); emailErr.textContent = 'Please enter your email address'; setFs(emailEl,'err'); }
      return false;
    }

    if(!/^[^\s@]+@[^\s@]+\.[^\s@]{2,}$/.test(val)){
      if(doShow){ showErr('slErr-email', true); emailErr.textContent = 'Please enter a valid email address'; setFs(emailEl,'err'); }
      return false;
    }

    var parts  = val.split('@');
    var prefix = parts[0];
    var domain = parts[1];

    if(TYPO_MAP[domain] && hintEl){
      var hintSpan = hintEl.querySelector('span');
      if(hintSpan) hintSpan.textContent = 'Did you mean ' + prefix + '@' + TYPO_MAP[domain] + '? Tap to fix';
      hintEl.style.display = 'flex';
      // Remove tick while domain hint is showing
      var emailField = emailEl.closest ? emailEl.closest('.sl-field') : null;
      if(emailField) emailField.removeAttribute('data-filled');
      hintEl.onclick = function(){
        emailEl.value = prefix + '@' + TYPO_MAP[domain];
        hintEl.style.display = 'none';
        applyFilled(emailEl, true);
        setFs(emailEl,'ok');
        showErr('slErr-email', false);
      };
    }

    if(DISPOSABLE_DOMAINS.indexOf(domain) > -1){
      if(doShow){ showErr('slErr-email', true); emailErr.textContent = 'Please use a personal email address вЂ” temporary inboxes aren\'t accepted'; setFs(emailEl,'err'); }
      return false;
    }

    if(ROLE_PREFIXES.indexOf(prefix) > -1){
      if(doShow){ showErr('slErr-email', true); emailErr.textContent = 'Please use your personal email address rather than a shared inbox'; setFs(emailEl,'err'); }
      return false;
    }

    if(doShow){ showErr('slErr-email', false); setFs(emailEl,'ok'); }
    return !TYPO_MAP[domain];
  }

  function applyFilled(el, onBlur){
    if(!el) return;
    var filled = el.value && el.value.trim() !== '';
    el.classList.toggle('sl-filled', !!filled);
    var field = el.closest ? el.closest('.sl-field') : null;
    if(!field){ var p = el.parentNode; while(p){ if(p.classList && p.classList.contains('sl-field')){ field=p; break; } p=p.parentNode; } }
    if(field){
      // Only show tick on blur (onBlur=true) or polling, and never when errors/warnings present
      var hasErr  = el.classList.contains('sl-err');
      var hasWarn = field.querySelector('.sl-phone-digits-hint.sl-warn') !== null;
      var hasEmailHint = field.querySelector('[id$="DomainHint"]') && field.querySelector('[id$="DomainHint"]').style.display !== 'none';

      if(showTick) field.setAttribute('data-filled','');
      else field.removeAttribute('data-filled');
    }
  }

function runPrefill () {
  var params = new URLSearchParams(window.location.search);
  function get (key) { return params.get(key); }

  window.slPrefilling = true;

  var type = get('type');
  if (type === 'A' || type === 'B') {
    var radio = document.querySelector('input[name="enquiryType"][value="' + type + '"]');
    if (radio) { radio.checked = true; radio.dispatchEvent(new Event('change')); }
  }

  var ref = get('ref');
  if (ref) {
    var refEl = document.getElementById('slAptRef');
    if (refEl) {
      if (!get('type')) {
        var radioA = document.querySelector('input[name="enquiryType"][value="A"]');
        if (radioA) { radioA.checked = true; radioA.dispatchEvent(new Event('change')); }
      }
      refEl.value = ref;
      applyFilled(refEl);
      setTimeout(function () { applyFilled(refEl); }, 50);
    }
  }

  var city = get('city');
  if (city) {
    var cityEl = document.getElementById('slCity');
    if (cityEl) { cityEl.value = city; cityEl.dispatchEvent(new Event('change')); applyFilled(cityEl); }
  }

  var apt = get('apt');
  if (apt) {
    var aptEl = document.getElementById('slAptType');
    if (aptEl) { aptEl.value = apt; aptEl.dispatchEvent(new Event('change')); applyFilled(aptEl); }
  }

  var name = get('name');
  if (name) {
    var nameEl = document.getElementById('slFullName');
    if (nameEl) { nameEl.value = name; applyFilled(nameEl); }
  }

  var email = get('email');
  if (email) {
    var emailEl = document.getElementById('slEmail');
    if (emailEl) { emailEl.value = email; applyFilled(emailEl); }
  }

  var checkin  = get('checkin');
  var checkout = get('checkout');
  if (checkin || checkout) {
    var attempts = 0;
    var wait = setInterval(function () {
      attempts++;
      var fpIn  = window._fpIn;
      var fpOut = window._fpOut;
      if ((fpIn && fpOut) || attempts > 20) {
        clearInterval(wait);
        if (checkin  && fpIn)  fpIn.setDate(checkin,  true);
        if (checkout && fpOut) fpOut.setDate(checkout, true);
      }
    }, 150);
  }

  window.slPrefilling = false;

  var uniParam = get('university');
  if (uniParam) {
    var uniEl = document.getElementById('slUniversity');
    if (uniEl) {
      var matched = false;
      for (var i = 0; i < uniEl.options.length; i++) {
        if (uniEl.options[i].value.toLowerCase() === uniParam.toLowerCase()) {
          uniEl.value = uniEl.options[i].value;
          applyFilled(uniEl);
          matched = true;
          break;
        }
      }
      if (!matched) {
        uniEl.value = 'Other';
        uniEl.dispatchEvent(new Event('change'));
        applyFilled(uniEl);
        var otherUniEl = document.getElementById('slOtherUni');
        if (otherUniEl) { otherUniEl.value = uniParam; applyFilled(otherUniEl); }
      }
    }
  }

  if ((type === 'A' || type === 'B') && typeof scrollToNextField === 'function') {
    setTimeout(function () { scrollToNextField(type === 'A'); }, 500);
  }
}

    var nationality=(document.getElementById('slNationality')?document.getElementById('slNationality').value:'').toLowerCase().trim();
