/* ── share/back-button support: mirror the active filters into the URL ── */
function writeUrlState() {
    if (!userTouched || !window.history || !history.replaceState) return;
    const params = new URLSearchParams(window.location.search);
    ['q','type','area','amenity','sort'].forEach(k => params.delete(k));
    if (searchQuery.trim()) params.set('q', searchQuery.trim());
    if (selectedBedrooms.length) params.set('type', selectedBedrooms.join(','));
    if (selectedAreas.length) params.set('area', selectedAreas.join(','));
    if (selectedAmenities.length) params.set('amenity', selectedAmenities.join(','));
    if (priceMin > priceBoundMin || priceMax < priceBoundMax) params.set('price', priceMin + '-' + priceMax);
    const sortVal = document.getElementById('sortSelect').value;
    if (sortVal !== 'price-low') params.set('sort', sortVal);
    const qs = params.toString();
    history.replaceState(null, '', window.location.pathname + (qs ? '?' + qs : '') + window.location.hash);
}

async function loadListings() {
  const { type, value } = PAGE_CONFIG.filter;
  const BLOB_URL = 'https://dnamuaiofr5we69r.public.blob.vercel-storage.com/listings.json';
  const cacheKey = `sl_listings_${type}_${value}`;

  const noCache = new URLSearchParams(window.location.search).get('nocache');
  const cached = noCache ? null : localStorage.getItem(cacheKey);
  const cachedTime = parseInt(localStorage.getItem(cacheKey + '_time') || '0', 10);
  const cacheAge = Date.now() - cachedTime;
  const CACHE_FRESH_MS = 5 * 60 * 1000;

  function bootFromData() {
      currentListings = [...listings];
      populateAreaFilter();
      urlTypes.forEach(t => {
          if (bedLabelsLong[t] && !selectedBedrooms.includes(t)) {
              selectedBedrooms.push(t);
              const cb = document.querySelector('#bedroomCheckboxList input[value="' + t + '"]');
              if (cb) cb.checked = true;
          }
      });
      updateBedroomDropdownLabel();
      urlAmenities.forEach(a => { if (!selectedAmenities.includes(a)) selectedAmenities.push(a); });
      if (urlQuery) { searchQuery = urlQuery; document.getElementById('searchInput').value = urlQuery; }
      if (urlSort && document.querySelector(`#sortSelect option[value="${urlSort}"]`)) {
          document.getElementById('sortSelect').value = urlSort;
      }
      initPriceSlider();
      renderAmenityPills();
      applyFilters();
  }

  if (cached) {
    try {
      const cachedListings = JSON.parse(cached);
      if (cachedListings.length > 0) {
        listings = cachedListings;
        bootFromData();
      }
    } catch(e) {
      localStorage.removeItem(cacheKey);
      localStorage.removeItem(cacheKey + '_time');
    }
  }

  if (cached && cacheAge < CACHE_FRESH_MS) {
    return;
  }

  try {
    const data = await (await fetch(BLOB_URL, { cache: 'no-store' })).json();
    const fresh = (data || []).filter(l => {
      if (PAGE_CONFIG.filter.type === 'city') return !l.hidden && l.city?.toLowerCase() === PAGE_CONFIG.filter.value.toLowerCase();
      if (PAGE_CONFIG.filter.type === 'area') return !l.hidden && l.location?.toLowerCase() === PAGE_CONFIG.filter.value.toLowerCase();
      return !l.hidden;
    }).map((l, i) => ({
      id: i + 1,
      mondayRef: l.mondayRef,
      buildingId: l.buildingId || l.mondayRef,
      title: l.title,
      location: l.location || '',
      bedroomType: l.bedroomType || '',
      weekly: l.weekly || 0,
      size: l.size || '',
      amenities: mapAmenities(l.amenities),
      url: l.url || '',
      imageUrl: l.imageUrl || '',
      imageUrl2: l.imageUrl2 || '',
      popular: l.popular || false,
      refurb: l.refurb || false,
      view: l.view || false,
      minstay90: l.minstay90 || false,
      comingSoon: l.comingSoon || false,
      poa: l.poa || false,
    }));

    if (fresh.length > 0) {
      try {
        localStorage.setItem(cacheKey, JSON.stringify(fresh));
        localStorage.setItem(cacheKey + '_time', Date.now().toString());
      } catch(e) {}

      if (JSON.stringify(fresh) !== JSON.stringify(listings)) {
        listings = fresh;
        bootFromData();
      }
    }
  } catch(e) {
    console.error('Failed to load fresh listings:', e);
    if (!cached) {
      listings = [];
      currentListings = [];
      document.getElementById('listingsGrid').innerHTML =
        '<div class="no-results" style="grid-column:1/-1"><h3>Unable to load apartments</h3>' +
        '<p>Something went wrong on our side.</p>' +
        '<button class="retry-btn" onclick="SLAF.retryLoad()">Try again</button>' +
        '<button type="button" class="enquiry-link" data-sl-open-enq>Or make an enquiry and we will send you a curated list</button></div>';
    }
  }
}

/* ── dual-thumb price range slider (state is always weekly £) ── */
const drMin = document.getElementById('priceMinRange');
const drMax = document.getElementById('priceMaxRange');
const drFill = document.getElementById('priceRangeFill');
const drLabel = document.getElementById('priceRangeLabel');
let priceDebounce;

/* log scale: slider positions 0-1000 map exponentially to £, so the busy
   low end of the market gets most of the track instead of a fifth of it */
const POS_MAX = 1000;
function posToPrice(p) {
    const ratio = Math.log(priceBoundMax / Math.max(priceBoundMin, 1));
    const v = Math.max(priceBoundMin, 1) * Math.exp(p / POS_MAX * ratio);
    return Math.min(priceBoundMax, Math.max(priceBoundMin, Math.round(v / 25) * 25));
}
function priceToPos(v) {
    const ratio = Math.log(priceBoundMax / Math.max(priceBoundMin, 1));
    return Math.round(Math.log(Math.max(v, priceBoundMin) / Math.max(priceBoundMin, 1)) / ratio * POS_MAX);
}

function updatePriceUi() {
    if (!drMin || !drFill) return;
    drFill.style.left = (priceToPos(priceMin) / 10) + '%';
    drFill.style.right = (100 - priceToPos(priceMax) / 10) + '%';
    const unit = priceMode === 'weekly' ? 'per week' : 'per month';
    const maxTxt = (priceCapped && priceMax >= priceBoundMax) ? formatPrice(priceMax) + '+' : formatPrice(priceMax);
    drLabel.textContent = `${PAGE_CONFIG.currency}${formatPrice(priceMin)} to ${PAGE_CONFIG.currency}${maxTxt} ${unit}`;
    const btnLabel = document.getElementById('priceDropdownLabel');
    if (btnLabel) btnLabel.textContent = `${PAGE_CONFIG.currency}${formatPrice(priceMin)} to ${PAGE_CONFIG.currency}${maxTxt}`;
    const groupLabel = document.getElementById('priceGroupLabel');
    if (groupLabel) groupLabel.textContent = priceMode === 'weekly' ? 'Weekly Price:' : 'Monthly Price:';
}

let priceCapped = false;   /* slider top = 95th percentile so one outlier can't squash the range */
function initPriceSlider() {
    if (!drMin || !listings.length) return;
    const values = listings.map(poaPrice).filter(v => v > 0).sort((a, b) => a - b);
    const lo = Math.floor(values[0] / 25) * 25;
    /* full range by default - the slider's top covers the most expensive listing */
    let hi = Math.ceil(values[values.length - 1] / 50) * 50;
    if (hi <= lo) hi = lo + 50;
    priceCapped = false;
    const firstInit = priceBoundMax === 0;
    priceBoundMin = lo; priceBoundMax = hi;
    [drMin, drMax].forEach(r => { r.min = 0; r.max = POS_MAX; r.step = 1; });
    if (firstInit) {
        priceMin = lo; priceMax = hi;
        if (urlPrice) {
            priceMin = Math.max(lo, Math.min(urlPrice[0], hi));
            priceMax = Math.max(priceMin, Math.min(urlPrice[1], hi));
        }
    } else {
        priceMin = Math.max(lo, Math.min(priceMin, hi));
        priceMax = Math.max(priceMin, Math.min(priceMax, hi));
    }
    drMin.value = priceToPos(priceMin); drMax.value = priceToPos(priceMax);
    updatePriceUi();
}

function onPriceInput() {
    let a = parseInt(drMin.value, 10), b = parseInt(drMax.value, 10);
    /* keep a minimum gap so the two thumbs can never stack - stacked thumbs
       are unrecoverable on iOS (the top input swallows every touch) */
    const GAP = 15;
    if (b - a < GAP) {
        if (this === drMin) { a = Math.max(0, b - GAP); drMin.value = a; }
        else { b = Math.min(POS_MAX, a + GAP); drMax.value = b; }
        if (b - a < GAP) { a = Math.max(0, b - GAP); drMin.value = a; }
    }
    priceMin = (a <= 0) ? priceBoundMin : posToPrice(a);
    priceMax = (b >= POS_MAX) ? priceBoundMax : posToPrice(b);
    updatePriceUi();
    clearTimeout(priceDebounce);
    priceDebounce = setTimeout(() => { userTouched = true; applyFilters(); renderAmenityPills(); }, 200);
}
if (drMin) { drMin.addEventListener('input', onPriceInput); drMax.addEventListener('input', onPriceInput); }

function populateAreaFilter() {
    const locationSet = new Set();
    listings.forEach(listing => {
        listing.location.split(',').map(p => p.trim()).filter(Boolean).forEach(p => locationSet.add(p));
    });
    const areas = [...locationSet].sort();
    const checkboxList = document.getElementById('areaCheckboxList');
    if (checkboxList) {
        checkboxList.innerHTML = areas.map(area => `
            <label style="display:flex;align-items:center;justify-content:flex-start;gap:10px;padding:8px 14px;cursor:pointer;font-size:14px;color:#333;text-align:left;transition:background 0.15s;" onmouseover="this.style.background='#f9fafb'" onmouseout="this.style.background='white'">
                <input type="checkbox" value="${escHtml(area)}" onchange="SLAF.onAreaCheckboxChange(this)" style="width:15px;height:15px;cursor:pointer;accent-color:#2563eb;">
                ${escHtml(area)}
            </label>`).join('');
    }
    /* Inbound ?area= links take priority; defaults apply only when none given */
    if (urlAreas.length) {
        selectedAreas = [];
        urlAreas.forEach(ua => {
            const match = areas.find(a => a.toLowerCase() === ua || a.toLowerCase().replace(/\s+/g, '-') === ua);
            if (match && !selectedAreas.includes(match)) selectedAreas.push(match);
        });
    } else if (PAGE_CONFIG.defaultAreas && PAGE_CONFIG.defaultAreas.length > 0) {
        selectedAreas = PAGE_CONFIG.defaultAreas.filter(a => areas.includes(a));
    }
    if (checkboxList) {
        checkboxList.querySelectorAll('input[type=checkbox]').forEach(cb => {
            cb.checked = selectedAreas.includes(cb.value);
        });
    }
    updateAreaDropdownLabel();
}

function onAreaCheckboxChange(checkbox) {
    userTouched = true;
    const area = checkbox.value;
    if (checkbox.checked) { if (!selectedAreas.includes(area)) selectedAreas.push(area); }
    else { selectedAreas = selectedAreas.filter(a => a !== area); }
    updateAreaDropdownLabel();
    applyFilters();
    renderAmenityPills();
}

function renderAmenityPills(){
    const container = document.getElementById('amenityFilterContainer');
    if(!container.dataset.expanded) container.dataset.expanded = 'false';
    const sortedKeys = Object.keys(amenityPriority).sort((a,b)=>amenityPriority[a]-amenityPriority[b]);
    const isDesktop = window.innerWidth > 768;
    const maxToShow = isDesktop ? 9 : 3;   /* mobile: 3 pills + "+x more" (+ reset) */
    const expanded = container.dataset.expanded === 'true';
    let keysToShow = expanded ? sortedKeys : sortedKeys.slice(0, maxToShow);
    let hiddenCount = sortedKeys.length - maxToShow;
    let html = keysToShow.map(key=>{
        const a = amenityIcons[key];
        const isSelected = selectedAmenities.includes(key);
        return `<button type="button" class="amenity-pill ${isSelected?'selected':''}" data-value="${key}" aria-pressed="${isSelected}"><i class="${a.icon}"></i> ${a.label}</button>`;
    }).join('');
    if(sortedKeys.length > maxToShow) {
        html += expanded
            ? `<button type="button" class="amenity-pill amenity-show-more">Show less</button>`
            : `<button type="button" class="amenity-pill amenity-show-more">+${hiddenCount} more</button>`;
    }
    const areaSelected = selectedAreas.length > 0;
    const searchActive = searchQuery && searchQuery.trim().length > 0;
    const bedroomSelected = selectedBedrooms.length > 0;
    if(selectedAmenities.length > 0 || areaSelected || searchActive || bedroomSelected) {
        html += `<button type="button" class="amenity-pill amenity-reset"><i class="fa-light fa-xmark"></i> Reset</button>`;
    }
    container.innerHTML = html;
    container.querySelectorAll('.amenity-pill:not(.amenity-show-more):not(.amenity-reset)').forEach(pill=>{
        pill.addEventListener('click', ()=>{
            userTouched = true;
            const value=pill.dataset.value;
            if(selectedAmenities.includes(value)) selectedAmenities=selectedAmenities.filter(a=>a!==value);
            else selectedAmenities.push(value);
            applyFilters(); renderAmenityPills();
        });
    });
    const resetBtn = container.querySelector('.amenity-reset');
    if(resetBtn) {
        resetBtn.addEventListener('click', () => {
            userTouched = true;
            selectedAmenities=[]; selectedBedrooms=[]; searchQuery=''; selectedAreas=[];
            priceMin = priceBoundMin; priceMax = priceBoundMax;
            if (drMin) { drMin.value = priceToPos(priceMin); drMax.value = priceToPos(priceMax); updatePriceUi(); }
            document.getElementById('searchInput').value='';
            document.getElementById('areaSelect').value='all';
            document.querySelectorAll('#bedroomCheckboxList input[type=checkbox]').forEach(cb=>cb.checked=false);
            updateBedroomDropdownLabel();
            document.querySelectorAll('#areaCheckboxList input[type=checkbox]').forEach(cb=>cb.checked=false);
            updateAreaDropdownLabel();
            applyFilters(); renderAmenityPills();
        });
    }
    const showMoreBtn = container.querySelector('.amenity-show-more');
    if(showMoreBtn) {
        showMoreBtn.addEventListener('click', function() {
            container.dataset.expanded = container.dataset.expanded === 'true' ? 'false' : 'true';
            renderAmenityPills();
        });
    }
}

function formatPrice(weeklyPrice) {
    if (priceMode === 'monthly') {
        const monthly = weeklyPrice * (52 / 12);
        const rounded = Math.round(monthly / 5) * 5;
        return rounded.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",");
    }
    return weeklyPrice.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",");
}

function getPriceText() { return priceMode === 'weekly' ? 'per week' : 'per month'; }

function getSizeValue(size){
    if(typeof size === "string" && size.includes("-")) return parseInt(size.split("-")[1], 10);
    return parseInt(size, 10);
}

function generateSchema(listing) {
    const bedrooms = listing.bedroomType==='studio'?0:listing.bedroomType==='1bed'?1:listing.bedroomType==='2bed'?2:listing.bedroomType==='3bed'?3:listing.bedroomType==='4bed'?4:0;
    const prettyType = bedLabelsLong[listing.bedroomType] || 'Apartment';
    const schema = {
        "@context":"https://schema.org","@type":"Apartment",
        "name":`${listing.title} - ${prettyType} apartment`,
        "description":`Luxury furnished ${prettyType.toLowerCase()} student apartment in ${PAGE_CONFIG.filter.value} with modern amenities`,
        "numberOfRooms":bedrooms,
        "floorSize":{"@type":"QuantitativeValue","value":listing.size,"unitCode":"MTK"},
        "address":{"@type":"PostalAddress","addressLocality":PAGE_CONFIG.filter.value,"addressRegion":PAGE_CONFIG.schemaRegion || PAGE_CONFIG.filter.value,"addressCountry":PAGE_CONFIG.schemaCountry || 'GB'},
        "amenityFeature":listing.amenities.map(amenity=>({
            "@type":"LocationFeatureSpecification","name":amenityIcons[amenity]?.label||amenity
        })).filter(a=>a.name),
        "image":listing.imageUrl,"url":listing.url
    };
    if(!listing.poa && !listing.comingSoon){
        schema.offers={"@type":"Offer","price":listing.weekly,"priceCurrency":"GBP",
            "availability":"https://schema.org/InStock",
            "priceSpecification":{"@type":"UnitPriceSpecification","price":listing.weekly,"priceCurrency":"GBP","unitText":"WEEK"}};
    }
    return schema;
}

function buildAmenityRow(listing) {
    const sortedAmenities = [...listing.amenities].sort((a,b)=>(amenityPriority[getAmenityGroup(a)]||99)-(amenityPriority[getAmenityGroup(b)]||99));
    const displayAmenities = [...sortedAmenities];
    const sizeValue = getSizeValue(listing.size);
    const bedType = listing.bedroomType;
    if((bedType==='studio'&&sizeValue>35)||(bedType==='1bed'&&sizeValue>55)||(bedType==='2bed'&&sizeValue>90)||(bedType==='3bed'&&sizeValue>120)||(bedType==='4bed'&&sizeValue>150)){
        displayAmenities.push('super-spacious-home');
    } else if((bedType==='studio'&&sizeValue>30)||(bedType==='1bed'&&sizeValue>50)||(bedType==='2bed'&&sizeValue>80)||(bedType==='3bed'&&sizeValue>110)||(bedType==='4bed'&&sizeValue>140)){
        displayAmenities.push('spacious-home');
    }
    return displayAmenities;
}

        ${extraCount>0?`<button type="button" class="amenity-more">+${extraCount} more</button>`:''}
    </div>`;

    const secondaryImg = SUPPORTS_HOVER && listing.imageUrl2

function updateResultsCount() {
    const el = document.getElementById('resultsCount');
    if (!el) return;
    const total = currentListings.length;
    if (!listings.length || total === 0) { el.textContent = ''; return; }
    /* one line: count + area context + clear link (search bypasses areas, so hide it then) */
    const areaClause = (PAGE_CONFIG.areaNoticeText && selectedAreas.length > 0 && !searchQuery.trim())
        ? ` ${escHtml(PAGE_CONFIG.areaNoticeText)} - <button type="button" onclick="SLAF.clearAreas()">${escHtml(PAGE_CONFIG.areaNoticeClearText || 'clear')}</button>`
        : '';
    el.innerHTML = `Showing <b>${Math.min(itemsToShow, total)}</b> of <b>${total}</b> apartments${areaClause}`;
}

function bindAmenityMore(scope) {
    (scope || document).querySelectorAll('.amenity-more').forEach(btn => {
        if (btn.dataset.bound) return;
        btn.dataset.bound = '1';
        btn.addEventListener('click', function toggleAmenities() {
            const container = btn.parentElement;
            const listingId = container.dataset.id;
            const expanded = container.dataset.expanded === "true";
            const listing = listings.find(l => l.id == listingId);
            if (!listing) return;
            const sortedAmenities = buildAmenityRow(listing);
            let html;
            if(expanded){
                const vis=sortedAmenities.slice(0,3); const ec=sortedAmenities.length-3;
                html=vis.map(key=>{const a=amenityIcons[key];return a?`<span class="amenity" title="${a.label}"><i class="${a.icon}"></i> ${a.label}</span>`:''}).join('');
                if(ec>0) html+=`<button type="button" class="amenity-more">+${ec} more</button>`;
                container.dataset.expanded="false";

                html=sortedAmenities.map(key=>{const a=amenityIcons[key];return a?`<span class="amenity" title="${a.label}"><i class="${a.icon}"></i> ${a.label}</span>`:''}).join('');

        });
    });
}

function renderListings(listingsToRender){
    const grid = document.getElementById('listingsGrid');
    if(listingsToRender.length===0){
        grid.innerHTML='<div class="no-results" style="grid-column:1/-1"><h3>No apartments found</h3><p>Try adjusting your filters</p></div>';
        updateResultsCount();
        return;
    }

    /* ONE combined ItemList schema (not per-card) keeps inline scripts to 1 */
    const visibleListings = listingsToRender.slice(0, itemsToShow);
    const combinedSchema = {
      "@context": "https://schema.org",
      "@type": "ItemList",
      "itemListElement": visibleListings.map((listing, i) => ({
        "@type": "ListItem",
        "position": i + 1,
        "item": generateSchema(listing)
      }))
    };
    const schemaMarkup = `<script type="application/ld+json">${JSON.stringify(combinedSchema)}<\/script>`;

    grid.innerHTML = schemaMarkup + visibleListings.map((l, i) => buildCard(l, i)).join('') + loadMoreMarkup();
    updateResultsCount();
    updateShortlistBar();
    bindAmenityMore(grid);
}

/* Append only the new cards - keeps scroll position, expanded rows, and
   already-decoded images intact instead of rebuilding the whole grid */
function loadMore() {
    const grid = document.getElementById('listingsGrid');
    const prev = itemsToShow;
    itemsToShow += PAGE_SIZE;
    const newCards = currentListings.slice(prev, itemsToShow);
    const oldFooter = grid.querySelector('.load-more-container');
    if (oldFooter) oldFooter.remove();
    grid.insertAdjacentHTML('beforeend', newCards.map(l => buildCard(l)).join('') + loadMoreMarkup());
    updateResultsCount();
    bindAmenityMore(grid);
}

function poaPrice(l) {
    return l.poa ? (l.bedroomType==='studio'?1000:l.bedroomType==='1bed'?1500:l.bedroomType==='2bed'?2500:l.bedroomType==='3bed'?3500:5000) : l.weekly;
}

function sortListings(sortType){
    switch(sortType){
        case 'quality': currentListings.sort((a,b)=>{
            const aHigh = a.amenities.includes('ultra-high-end');
            const bHigh = b.amenities.includes('ultra-high-end');
            if(aHigh && !bHigh) return -1;
            if(!aHigh && bHigh) return 1;
            if(aHigh && bHigh) {
                if(a.popular && !b.popular) return -1;
                if(!a.popular && b.popular) return 1;
            }
            return poaPrice(a) - poaPrice(b);
        }); break;
        /* popular first, then stable price tiebreak (no random reshuffles) */
        case 'popular': currentListings.sort((a,b)=>{
            if(a.popular&&!b.popular)return -1;
            if(!a.popular&&b.popular)return 1;
            return poaPrice(a) - poaPrice(b);
        }); break;
        case 'price-low': currentListings.sort((a,b)=>poaPrice(a)-poaPrice(b)); break;
        case 'price-high': currentListings.sort((a,b)=>poaPrice(b)-poaPrice(a)); break;
        case 'size-small': currentListings.sort((a,b)=>getSizeValue(a.size)-getSizeValue(b.size)); break;
        case 'size-large': currentListings.sort((a,b)=>getSizeValue(b.size)-getSizeValue(a.size)); break;
    }
    renderListings(currentListings);
}

function applyFilters(){
    const searching = !!searchQuery.trim();
    currentListings=listings.filter(listing=>{
        if(searchQuery&&!listing.title.toLowerCase().includes(searchQuery.toLowerCase())&&!listing.location.toLowerCase().includes(searchQuery.toLowerCase())) return false;
        if(selectedBedrooms.length > 0 && !selectedBedrooms.includes(listing.bedroomType)) return false;
        /* explicit search wins over the area pre-filter, so a searched name or
           area outside the defaults still returns results */
        if(!searching && selectedAreas.length && !selectedAreas.some(area => listing.location.includes(area))) return false;
        if((priceMin > priceBoundMin || priceMax < priceBoundMax) && (poaPrice(listing) < priceMin || poaPrice(listing) > priceMax)) return false;
        if(selectedAmenities.length){
            return selectedAmenities.every(sel=>listing.amenities.some(la=>getAmenityGroup(la)===sel));
        }
        return true;
    });
    itemsToShow=PAGE_SIZE;
    writeUrlState();
    sortListings(document.getElementById('sortSelect').value);
}

function escapeRegex(s){return s.replace(/[.*+?^${}()|[\]\\]/g,'\\$&');}

function showAutocompleteSuggestions(query){
    if(!query||query.length<2){autocompleteDropdown.classList.remove('show');searchInput.setAttribute('aria-expanded','false');return;}
    const lowerQuery=query.toLowerCase();
    const titleMatches=[];const areaMatches=new Set();
    listings.forEach(listing=>{
        if(listing.title.toLowerCase().includes(lowerQuery)) titleMatches.push(listing.title);
        listing.location.split(',').map(p=>p.trim()).forEach(p=>{if(p.toLowerCase().includes(lowerQuery))areaMatches.add(p);});
    });
    const uniqueTitles=[...new Set(titleMatches)];const uniqueAreas=[...areaMatches];
    let html='';
    if(uniqueTitles.length>0){html+='<div class="autocomplete-category">Apartments</div>';uniqueTitles.slice(0,5).forEach(title=>{html+=`<div class="autocomplete-item title" role="option" data-value="${escHtml(title)}" data-type="title">${highlightMatch(title,query)}</div>`;});}
    if(uniqueAreas.length>0){html+='<div class="autocomplete-category">Areas</div>';uniqueAreas.slice(0,5).forEach(area=>{html+=`<div class="autocomplete-item area" role="option" data-value="${escHtml(area)}" data-type="area">${highlightMatch(area,query)}</div>`;});}
    if(html==='') html='<div class="autocomplete-no-results">No matches found</div>';
    autocompleteDropdown.innerHTML=html;autocompleteDropdown.classList.add('show');searchInput.setAttribute('aria-expanded','true');selectedSuggestionIndex=-1;
    autocompleteDropdown.querySelectorAll('.autocomplete-item').forEach(item=>{
        item.addEventListener('click',()=>{
            userTouched=true;
            if(item.dataset.type==='area'){
                /* picking an area ticks it in the Area dropdown instead of
                   fighting the pre-filter via free-text search */
                const area=item.dataset.value;
                if(!selectedAreas.includes(area)) selectedAreas.push(area);
                document.querySelectorAll('#areaCheckboxList input[type=checkbox]').forEach(cb=>{if(cb.value===area)cb.checked=true;});
                updateAreaDropdownLabel();

            }
            autocompleteDropdown.classList.remove('show');searchInput.setAttribute('aria-expanded','false');
            applyFilters();renderAmenityPills();
        });
    });
}

/* deferred load: fetch when the grid nears the viewport (or after 3s max).
   NOTE: this section starts display:none, so the observer cannot fire until
   it is revealed - the 3s fallback below does the real work and the data is
   ready by the time anyone clicks "View apartments near Marangoni". */
(function bootLoad() {
  var grid = document.getElementById('listingsGrid');
  if (!grid) { loadListings(); return; }

  var params = new URLSearchParams(window.location.search);
  if (params.has('type') || params.has('amenity') || params.has('area') || params.has('q') || params.has('sort')) {
    loadListings();
    return;
  }

  if (window.innerWidth <= 480 && grid.getBoundingClientRect().top < window.innerHeight * 2) {
    loadListings();
    return;
  }

  if (!('IntersectionObserver' in window)) {
    loadListings();
    return;
  }

  var loaded = false;
  var observer = new IntersectionObserver(function(entries) {
    if (entries[0].isIntersecting && !loaded) {
      loaded = true;
      observer.disconnect();
      loadListings();
    }
  }, { rootMargin: '600px 0px', threshold: 0.01 });

  observer.observe(grid);

  setTimeout(function() {
    if (!loaded) {
      loaded = true;
      observer.disconnect();
      loadListings();
    }
  }, 3000);
})();
