  /* ── rotating price in the value line (self-correcting: only one ever visible) ── */
  var INTERVAL=2500, FADE=550, EASE='ease-in-out';
  var reduce = window.matchMedia && window.matchMedia('(prefers-reduced-motion:reduce)').matches;
  var rot=document.getElementById('slBValRot');
  if(rot){
    var items=Array.prototype.slice.call(rot.querySelectorAll('.item'));
    if(items.length){
      var ri=0;
      /* renderVal() ALWAYS forces exactly one item visible and hides the rest,
         so the line can never "double up" even if timers drift in a background tab */
      function renderVal(fade){
        items.forEach(function(it,k){
          if(k===ri){
            it.style.display='inline';
            if(fade){ it.style.transition='none'; it.style.opacity='0'; void it.offsetWidth; it.style.transition='opacity '+FADE+'ms '+EASE; it.style.opacity='1'; }
            else { it.style.transition='none'; it.style.opacity='1'; }
          } else { it.style.transition='none'; it.style.display='none'; it.style.opacity='0'; }
        });
      }
      renderVal(false);
      if(items.length>1 && !reduce){
        setInterval(function(){
          if(document.hidden) return;                 /* don't animate while tab is hidden */
          var cur=items[ri]; cur.style.transition='opacity '+FADE+'ms '+EASE; cur.style.opacity='0';
          setTimeout(function(){ ri=(ri+1)%items.length; renderVal(true); }, FADE+20);
        }, INTERVAL);
        document.addEventListener('visibilitychange', function(){ if(!document.hidden){ renderVal(false); } });
      }
    }
  }
})();
