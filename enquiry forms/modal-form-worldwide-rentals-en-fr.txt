  var rot=document.getElementById('stlValRot');

    var items=Array.prototype.slice.call(rot.querySelectorAll('.item'));
    if(items.length){
      var ri=0;
      function renderVal(fade){

          if(k===ri){

      if(items.length>1 && !reduce){
        setInterval(function(){
          if(document.hidden) return;
          var cur=items[ri]; cur.style.transition='opacity '+FADE+'ms '+EASE; cur.style.opacity='0';
